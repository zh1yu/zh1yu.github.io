题七（二元LCG）

题目描述：

from secret import flag

assert flag[:5] == b'cazy{'
assert flag[-1:] == b'}'
flag = flag[5:-1]
assert (len(flag) == 24)


class my_LCG:
    def __init__(self, seed1, seed2):
        self.state = [seed1, seed2]
        self.n = getPrime(64)
        while 1:
            self.a = bytes_to_long(flag[:8])
            self.b = bytes_to_long(flag[8:16])
            self.c = bytes_to_long(flag[16:])
            if self.a < self.n and self.b < self.n and self.c < self.n:
                break

    def next(self):
        new = (self.a * self.state[-1] + self.b * self.state[-2] + self.c) % self.n
        self.state.append(new)
        return new


def main():
    lcg = my_LCG(getRandomInteger(64), getRandomInteger(64))
    print("data = " + str([lcg.next() for _ in range(5)]))
    print("n = " + str(lcg.n))

import itertools
def small_roots(f, bounds, m=1, d=None):
    if not d:
        d = f.degree()
    R = f.base_ring()
    N = R.cardinality()
    f /= f.coefficients().pop(0)
    f = f.change_ring(ZZ)
    G = Sequence([], f.parent())
    for i in range(m+1):
        base = N^(m-i) * f^i
        for shifts in itertools.product(range(d), repeat=f.nvariables()):
            g = base * prod(map(power, f.variables(), shifts))
            G.append(g)
    B, monomials = G.coefficient_matrix()
    monomials = vector(monomials)
    factors = [monomial(*bounds) for monomial in monomials]
    for i, factor in enumerate(factors):
        B.rescale_col(i, factor)
    B = B.dense_matrix().LLL()
    B = B.change_ring(QQ)
    for i, factor in enumerate(factors):
        B.rescale_col(i, 1/factor)
    H = Sequence([], f.parent().change_ring(QQ))
    for h in filter(None, B*monomials):
        H.append(h)
        I = H.ideal()
        if I.dimension() == -1:
            H.pop()
        elif I.dimension() == 0:
            roots = []
            for root in I.variety(ring=ZZ):
                root = tuple(R(root[var]) for var in f.variables())
                roots.append(root)
            return roots
    return []

### LCG算法的C语言实现及用途 LCG（线性同余生成器，Linear Congruential Generator）是一种经典的伪随机数生成算法。它通过一个递归公式生成一系列伪随机数，其形式为： \[ X_{n+1} = (a \cdot X_n + c) \mod m \] 其中，X0X0 是种子值，aa 是乘数，cc 是增量，mm 是模数。这些参数的选择直接影响生成的伪随机数的质量，包括周期长度和均匀性[^1]。 #### 参数选择的重要性 LCG 的性能高度依赖于参数 aa、cc 和 mm 的选择。如果参数选择不当，可能会导致生成的序列具有较短的周期或较差的统计特性。例如，当 \( c = 0 \) 时，算法退化为乘法同余法（Multiplicative Congruential Method, MCM），其周期长度通常更短[^2]。 #### C语言实现示例 以下是一个基于 C 语言的 LCG 实现示例： ```c #include <stdio.h> #include <stdlib.h> #define MODULUS 2147483647 // 模数 m #define MULTIPLIER 16807 // 乘数 a #define INCREMENT 0 // 增量 c unsigned long lcg(unsigned long seed) { return (MULTIPLIER * seed + INCREMENT) % MODULUS; } int main() { unsigned long seed = time(NULL); // 使用当前时间作为种子 int n = 10; // 生成 10 个随机数 printf("生成的伪随机数序列:\n"); for (int i = 0; i < n; i++) { seed = lcg(seed); printf("%lu\n", seed); } return 0; } ``` #### LCG算法的应用场景 LCG 算法因其简单性和高效性，在许多领域中得到了广泛应用： - **模拟与建模**：在需要大量伪随机数的蒙特卡罗模拟中，LCG 是一种常见的选择[^3]。 - **游戏开发**：用于生成不可预测的游戏事件或关卡设计。 - **加密学**：尽管 LCG 不适合直接用于现代加密，但在某些低安全性要求的场景中仍可发挥作用。 - **统计计算**：用于生成均匀分布的随机样本，支持各种统计分析任务。 #### 注意事项 虽然 LCG 算法简单易用，但其生成的伪随机数可能存在一定的相关性，尤其是在高维空间中。因此，在对随机性要求较高的场景中，可能需要考虑其他更先进的随机数生成算法，如梅森旋转算法（Mersenne Twister）[^4]。
